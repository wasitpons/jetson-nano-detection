# jetson_runtime — Python dependencies.
#
# Install on Jetson Nano (Python 3.6.9, JetPack 4.6.x):
#     pip3 install --user -r requirements.txt
#     pip3 install --user 'pycuda<2022.0'         # needs CUDA toolkit; see below
#
# Install on dev box (macOS / Linux, Python ≥ 3.7) — for smoke testing only:
#     pip3 install -r requirements.txt
#     pip3 install 'opencv-python>=4.5'           # see below
#
# ---------------------------------------------------------------------------
# DELIBERATELY NOT IN THIS FILE
# ---------------------------------------------------------------------------
#
#   tensorrt    — ships with JetPack. Don't pip-install it.
#
#   pycuda      — needs CUDA toolkit on PATH. Excluded so a dev box without
#                 CUDA doesn't fail this whole file. On Jetson:
#                     pip3 install --user 'pycuda<2022.0'
#
#   opencv-python — on Jetson, use the JetPack-provided cv2 (built with
#                 GStreamer + Tegra plugins; nvv4l2decoder, nvvidconv). Pip's
#                 opencv-python would shadow it and break hardware decode.
#                 On a dev box for smoke testing only:
#                     pip3 install 'opencv-python>=4.5'
#
# Version pins below cap older releases for Python 3.6 (Jetson). On newer
# Python the floors are raised so dev boxes get current-line packages.
# ---------------------------------------------------------------------------

# Backport: `dataclasses` is stdlib since Py3.7; on Py3.6 we need the
# PyPI backport. Detector/base.py + frame_buffer.py rely on @dataclass.
dataclasses; python_version < "3.7"

# Config loading. PyYAML 6.0.2 dropped Py3.6 (requires_python >=3.8).
PyYAML>=5.1,<6.0.2; python_version < "3.7"
PyYAML>=5.1;        python_version >= "3.7"

# Numerical core. numpy 1.20+ dropped Python 3.6 — 1.19.5 is the last release.
numpy>=1.19,<1.20; python_version < "3.7"
numpy>=1.21;       python_version >= "3.7"

# Process / system metrics (CPU%, RSS, thread count) for MetricsCollector.
# psutil 7.x still supports Py3.6.
psutil>=5.7

# Live TUI dashboard. rich 13+ dropped Py3.6 (rich 12.6.0 is last for Py3.6.3+).
# rich 10.x is plenty for our use (Live + Table + Layout + Panel).
rich>=10.0,<11;    python_version < "3.7"
rich>=12.0;        python_version >= "3.7"

# Optional but recommended — lets tools/_write_engine_metadata.py probe the
# ONNX input shape into engine_metadata.json. If absent, that field is null
# in the metadata and the rest of the pipeline still works.
# onnx>=1.11 dropped Py3.6 + arm64 Py3.6 wheels are scarce after 1.10.
onnx>=1.8,<1.11;   python_version < "3.7"
onnx>=1.12;        python_version >= "3.7"

# onnx pulls in protobuf; protobuf 4.x dropped Py3.6. Pin to the 3.x line on
# Jetson so pip doesn't resolve a wheel that refuses to install.
protobuf<4;        python_version < "3.7"
